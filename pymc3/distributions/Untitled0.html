    prec = pm.Wishart('prec', 100., prec_prior / 100., shape=(2, 2))

    start = pm.find_MAP(fmin=opt.fmin_powell)
    #start = {'prec': prec_prior}
    step = pm.Metropolis()
    trace = pm.sample(50000, step, start)
