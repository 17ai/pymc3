import pymc3 as pm

y = 2*X + 1

with pm.Model() as model_glm:
    pm.glm.glm("y~X", data)
    print("Finished GLM")
    start = pm.find_MAP()
    print("Finished MAP")
    print(start)
    #hess = pm.find_hessian(start)
    start['sd_log'] = np.array(0.)
    step = pm.NUTS(scaling=start, step_scale=0.01)
