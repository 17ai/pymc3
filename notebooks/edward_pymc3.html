import tensorflow as tf
import edward as ed
from edward.models import Normal, Bernoulli, Empirical
import numpy as np

import matplotlib.pyplot as plt

def sigmoid(x):
    return 1 / (1 + np.exp(-x))

np.random.seed(123)
D = 54
N = 581012
T = 100

true_beta = np.random.randn(D)

x_data = np.random.randn(N, D)
p = sigmoid(np.dot(x_data, true_beta))
y_data = np.array([np.random.binomial(1, i) for i in p])

#!/usr/bin/env python
import numpy as np
import pystan

# Define the model.
MU_X = 0.0
MU_Y = 3.0
SIGMA_X = 1.0
SIGMA_Y = 2.0
RHO = 0.5

stan_model = """\
parameters {{
  vector[2] z;
}}

model {{
  matrix[2,2] sigma;
  vector[2] mu;

  mu[1] <- {MU_X};
  mu[2] <- {MU_Y};
  sigma[1][1] <- {SIGMA_X} * {SIGMA_X};
  sigma[1][2] <- {RHO} * {SIGMA_X} * {SIGMA_Y};
  sigma[2][1] <- {RHO} * {SIGMA_X} * {SIGMA_Y};
  sigma[2][2] <- {SIGMA_Y} * {SIGMA_Y};

  z ~ multi_normal(mu, sigma);
}}""".format(MU_X=MU_X, MU_Y=MU_Y, SIGMA_X=SIGMA_X, SIGMA_Y=SIGMA_Y, RHO=RHO)
print("STAN model:\n\n%s" % stan_model)

# Compile it.
model = pystan.StanModel(model_code=stan_model, model_name='TWOD_Gaussian', verbose=True)

# Sample
alg = 'NUTS'
rng = np.random.RandomState(seed=5678)
fit = model.sampling(warmup=100000,
                     iter=1000000,
                     verbose=True,
                     n_jobs=4,
                     chains=4,
                     seed=rng,
                     algorithm=alg)

# Print stats.
print(' ')
print("pystan version:", pystan.__version__)
print('algorithm:', alg)
print(fit)
z = fit.extract()['z']
print("mean  :", np.mean(z, axis=0))
print("stddev:", np.std(z, axis=0))
print("corr  :", np.corrcoef(z.T)[0, 1])
